	t = Object(this)
	len = t.length >>> 0
	result = []
	for i in [0...len]
		if t[i] instanceof Array
			result[i] = t[i].mult(arr[i])
		else
			result[i] ?= t[i] * arr[i]
	return result

    map

  • ¶

    map is an implementation of the EMCA polyfill for map with one exception—-it iterates over 2d-arrays.

    Array::map = (f, args...) ->
	t = Object(this)
	len = t.length >>> 0

	if typeof f isnt "function"
		throw new TypeError()

	result = []
	thisArg = if arguments.length >= 2 then arguments[1] else undefined
	for i in [0...len]
		if t[i] instanceof Array
			result[i] = t[i].map f, args...
		else
			result[i] ?= f.call thisArg, t[i], i, t
